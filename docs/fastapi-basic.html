#import all neccessary libraries

from fastapi import FastAPI

import nest_asyncio

from uvicorn import Config, Server

nest_asyncio.apply()

#connect the FastAPI

app = FastAPI()

#create a route for get method

@app.get("/")
def read_root():
    return {"message": "Hello from FastAPI in JupyterLab!"}

# Run FastAPI server

config = Config(
    app      =app,
    host     ="127.0.0.1",
    port     =8000,
    log_level="info"
)

server = Server(config)

server.run()
