def home():
    screen = turtle.Screen()
    t = turtle.Turtle()

        t.forward(100)
        t.right(90)
    turtle.done()
    return turtle

print(home())

import yaml

# Read YAML
yaml_data = """
name: Alice
age: 25
skills:
  - Python
  - Machine Learning
"""
data = yaml.safe_load(yaml_data)
print(data)

# Write YAML
with open("output.yaml", "w") as file:
    yaml.dump(data, file)
