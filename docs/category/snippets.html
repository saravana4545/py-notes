def cric_score():

    runs = [15000, 10500, 14000]

    plt.bar(players, runs)

    plt.ylabel("Runs")
    plt.show()

import namegenerator

import uuid

def name_gen():
    random_name = namegenerator.gen()

    random_name = namegenerator.gen()
    unique_name = f"{random_name}-{uuid.uuid4().hex[:8]}"
    return unique_name

def startpy():
    print(name_gen())
