from datetime import datetime

def get_age(d):
    d1 = datetime.now()
    months = (d1.year - d.year) * 12 + d1.month - d.month

    year = int(months / 12)
    return year

age = get_age(datetime(1991, 1, 1))

age

Basic-Json

import json

class City:
    def toJSON(self):
        return json.dumps(self, default=lambda o: o.__dict__, sort_keys=True, indent=4)

me = City()
me.name = "Onur"
me.id = 1

print(me.toJSON()) 

{
    "id": 1,
    "name": "Onur"
}
