import qrcode

def make_qrcode():
    qr = qrcode.QRCode(
        version=1,
        error_correction=qrcode.constants.ERROR_CORRECT_L,
        box_size=10,
        border=4
    )
    qr.add_data("https://github.com/saravana4545/")
    qr.make(fit=True)
    img = qr.make_image(
        fill_color="black",
        back_color="white"

    img.save("qrcode.png")

    return img

import namegenerator

import uuid

def name_gen():
    random_name = namegenerator.gen()
    return random_name

def unique_id():
    random_name = namegenerator.gen()
    unique_name = f"{random_name}-{uuid.uuid4().hex[:8]}"
    return unique_name

def startpy():
    print(name_gen())
