def sum_of_digits(num):
    digit_sum = 0
    while num > 0:
        digit_sum += num % 10
        num //= 10
    return digit_sum

list = 4545
print(sum_of_digits(list))

#Finds and prints the smallest and largest numbers in the list.

def large_and_small_nums(num):
    max_num = max(num)
    min_num = min(num)
    return max_num, min_num

list = (10,25,45,50,24)
large_and_small_nums(list)

    a = a+b
    b = a-b
    a = a-b
    return f"After swap a={a},b={b}"

import json

class City:
    def toJSON(self):
        return json.dumps(self, default=lambda o: o.__dict__, sort_keys=True, indent=4)

me = City()
me.name = "Onur"
me.id = 1

print(me.toJSON())

    "id": 1,
    "name": "Onur"
