Basic-Json
Fri 10 January 2025 
import json

class City:
    def toJSON(self):
        return json.dumps(self, default=lambda o: o.__dict__, sort_keys=True, indent=4)

me = City()
me.name = "Onur"
me.id = 1

print(me.toJSON()) 

{
    "id": 1,
    "name": "Onur"
}
