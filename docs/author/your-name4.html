import namegenerator

import uuid

def name_gen():
    random_name = namegenerator.gen()

    random_name = namegenerator.gen()
    unique_name = f"{random_name}-{uuid.uuid4().hex[:8]}"
    return unique_name

def startpy():
    print(name_gen())
    print(unique_id())

    return string == string[::-1]
