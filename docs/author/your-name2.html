Basic-Json
Wed 20 November 2024 
import json

class City:
    def toJSON(self):
        return json.dumps(self, default=lambda o: o.__dict__, sort_keys=True, indent=4)

me = City()
me.name = "Onur"
me.id = 1

print(me.toJSON()) 

{
    "id": 1,
    "name": "Onur"
}
