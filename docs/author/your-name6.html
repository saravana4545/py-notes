import hashlib

# Original password

password = "Saro@2003"

# Hash the password

hashed_password = hashlib.sha256(password.encode()).hexdigest()

# Simulate user input and hash it

user_input = "Saro@2003"
hashed_input = hashlib.sha256(user_input.encode()).hexdigest()

# Compare hashes

if hashed_password == hashed_input:
    print("Password match!")
else:
    print("Incorrect password.")

#create a method called gcd()

def gcd(a, b):
    if b == 0:  
        return a
    else:
        return gcd(b, a % b)

#put the values for a and b by calling the method
