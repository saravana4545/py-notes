def is_palindrome(string):
    string = string.lower().replace(" ", "")
    return string == string[::-1]

text = input("Enter a string: ")
if is_palindrome(text):
    print(f'"{text}" is a palindrome.')
else:
    print(f'"{text}" is not a palindrome.')
